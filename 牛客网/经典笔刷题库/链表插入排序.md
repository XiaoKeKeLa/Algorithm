**链表插入排序**

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @return ListNode类
     */
    ListNode* insertionSortList(ListNode* head) {
        ListNode *p=head;
        vector<int> _t;
        if(head==NULL)
            return NULL;
        while(p!=NULL)
        {
            _t.push_back(p->val);
            p=p->next;
        }
        p=head;
        sort(_t.begin(),_t.end());
        for(int i=0;i<_t.size();i++)
        {
            p->val=_t[i];
            p=p->next;
        }
        return head;
    }
};
```
