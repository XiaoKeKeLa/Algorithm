**求目标值的区间**

```c++
class Solution {
public:
    /**
     * 
     * @param A int整型一维数组 
     * @param n int A数组长度
     * @param target int整型 
     * @return int整型vector
     */
    vector<int> searchRange(int* A, int n, int target) {
        vector<int> res;
        
        for(int i=0;i<n;i++)
        {
            
            if(A[i]==target)
            {
                res.push_back(i);
                int index=i;
                while(A[index+1]==target)
                {
                    index++;
                }
                res.push_back(index);
                break;
            }
        }
        
        if(res.empty())
        {
            res.push_back(-1);
            res.push_back(-1);
        }
        return res;
    }
};
```
